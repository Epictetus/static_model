      def define_association_methods
        if reflection_klass.respond_to?(:scoped)
          klass.module_eval "def \#{name}\n\#{reflection_klass}.scoped(:conditions => {:\#{foreign_key} => id})\nend\n"
        else
          klass.module_eval "def \#{name}\n\#{reflection_klass}.send(:find_all_by_\#{foreign_key}, id)\nend\n"
        end
      end
